# ============================================================================
# Installation Notes for UI Requirements
# ============================================================================
# 1. Install UI dependencies:
#    pip install -r requirements/ui.txt
#
# 2. Launch Streamlit app:
#    streamlit run app/streamlit_app.py
#    Access at: http://localhost:8501
#
# 3. Launch Gradio demo:
#    python app/gradio_app.py
#    Access at: http://localhost:7860
#
# 4. Streamlit app structure:
#    - Main app: app/streamlit_app.py
#    - 16 pages in app/pages/:
#      * 01_Home.py
#      * 02_Single_Prediction.py
#      * 03_Batch_Analysis.py
#      * 04_Model_Comparison.py
#      * 05_Overfitting_Dashboard.py
#      * 06_Model_Recommender.py
#      * 07_Parameter_Efficiency_Dashboard.py
#      * 08_Interpretability.py
#      * 09_Performance_Dashboard.py
#      * 10_Real_Time_Demo.py
#      * 11_Model_Selection.py
#      * 12_Documentation.py
#      * 13_Prompt_Testing.py
#      * 14_Local_Monitoring.py
#      * 15_IDE_Setup_Guide.py
#      * 16_Experiment_Tracker.py
#
# 5. Components:
#    - Prediction: app/components/prediction_component.py
#    - Overfitting monitor: app/components/overfitting_monitor.py
#    - LoRA config selector: app/components/lora_config_selector.py
#    - Ensemble builder: app/components/ensemble_builder.py
#    - Model selector: app/components/model_selector.py
#
# 6. Customization:
#    - Theme: app/assets/css/custom.css
#    - JavaScript: app/assets/js/custom.js
#    - Images: app/assets/images/
#
# 7. Session state:
#    - Managed in app/utils/session_manager.py
#    - Caching in app/utils/caching.py
#
# 8. Visualizations:
#    - Training curves: src/evaluation/visualizations/training_curves.py
#    - Confusion matrix: src/evaluation/visualizations/confusion_matrix.py
#    - Attention viz: src/evaluation/visualizations/attention_visualization.py
#    - LoRA weights: src/evaluation/visualizations/lora_weight_visualization.py
#
# 9. Interactive dashboards:
#    - Overfitting: monitoring/dashboards/wandb/overfitting_dashboard.json
#    - Parameter efficiency: monitoring/dashboards/wandb/parameter_efficiency_dashboard.json
#    - Training: monitoring/dashboards/wandb/training_dashboard.json
#
# 10. CLI with rich output:
#     - Rich console: tools/cli_helpers/rich_console.py
#     - Progress bars: tools/cli_helpers/progress_bars.py
#     - Interactive prompts: tools/cli_helpers/interactive_prompts.py
#
# Platform-specific notes:
# - Windows: Most packages work, some 3D viz may have issues
# - Linux: Full support
# - macOS: Full support
#
# Browser compatibility:
# - Chrome/Edge: Full support (recommended)
# - Firefox: Full support
# - Safari: Mostly supported
#
# Performance tips:
# - Use st.cache_data for expensive computations
# - Enable WebSocket compression
# - Use session state efficiently
# - Lazy load large visualizations
#
# Deployment:
# - Local: streamlit run app/streamlit_app.py
# - Streamlit Cloud: Free tier available
# - Hugging Face Spaces: See deployment/huggingface/
# - Docker: See deployment/docker/
#
# For detailed UI documentation:
# - Streamlit: https://docs.streamlit.io
# - Gradio: https://gradio.app/docs
# - See quickstart/demo_app.py for simple example
# ============================================================================
